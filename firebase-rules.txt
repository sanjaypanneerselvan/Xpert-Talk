// --- COPY ONLY THIS SECTION FOR FIRESTORE RULES ---
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isAuthorized() {
      return request.auth != null;
    }
    
    match /users/{userId} {
      allow read: if request.auth != null;
      allow write: if isAuthorized() && request.auth.uid == userId;
    }
    
    match /tasks/{taskId} {
      allow read, write: if isAuthorized();
    }
    
    match /chats/{chatId}/{document=**} {
      allow read, write: if isAuthorized() && chatId.contains(request.auth.uid);
    }
    
    match /group_messages/{msgId} {
      allow read, write: if isAuthorized();
    }
    
    match /ideas/{ideaId} {
      allow read, write: if isAuthorized();
    }
    
    match /moms/{momId} {
      allow read, write: if isAuthorized();
    }
    
    match /activity/{logId} {
      allow read: if isAuthorized();
      allow create: if request.auth != null;
    }
    
    match /notifications/{notifId} {
      allow read, update: if isAuthorized() && resource.data.userId == request.auth.uid;
      allow create: if isAuthorized();
    }
  }
}

  }
